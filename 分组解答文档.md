### 需要讲解的知识点

#### 1. MVVM 架构（重点）

> "我们的 App 采用了 Google 官方推荐的 **MVVM 架构**，分为三层：
> - **Model 层** — 数据（Room 数据库 + Entity）
> - **ViewModel 层** — 业务逻辑（CourseViewModel）
> - **View 层** — UI（Compose 界面）
>
> 这样分层的好处是：各层职责清晰，ViewModel 不持有 View 的引用，方便单元测试。"

**对应的代码结构图：**
```
View (Compose UI) → ViewModel (StateFlow) → Model (Room DB)
      |                      |                      |
  观察UI状态             管理业务逻辑           持久化数据
```

#### 2. Room 数据库（核心）

**`AppDatabase.kt`：**
```kotlin
@Database(
    entities = [Course::class, Schedule::class],
    version = 1,
    exportSchema = false
)
abstract class AppDatabase : RoomDatabase() {
    // 单例模式，确保整个 App 只有一个数据库实例
    companion object {
        @Volatile
        private var INSTANCE: AppDatabase? = null

        fun getDatabase(context: Context): AppDatabase {
            return INSTANCE ?: synchronized(this) {
                val instance = Room.databaseBuilder(
                    context.applicationContext,
                    AppDatabase::class.java,
                    "course_table_database"  // 数据库文件名
                )
                    .fallbackToDestructiveMigration()  // 版本升级时重建
                    .build()
                INSTANCE = instance
                instance
            }
        }
    }
}
```

**讲解要点：**
- `@Database` 注解声明这是一个 Room 数据库
- 使用 **单例模式** (`synchronized`) 确保线程安全
- `fallbackToDestructiveMigration()` 开发阶段方便，生产环境要用 Migration

#### 3. Entity 实体类

**`Course.kt`：**
```kotlin
@Entity(
    tableName = "courses",
    foreignKeys = [
        ForeignKey(
            entity = Schedule::class,
            parentColumns = ["id"],
            childColumns = ["scheduleId"],
            onDelete = ForeignKey.CASCADE  // 删除课表时级联删除课程
        )
    ],
    indices = [Index("scheduleId")]
)
data class Course(
    @PrimaryKey(autoGenerate = true) val id: Long = 0,
    val scheduleId: Long = 0,       // 外键，关联到课表
    val name: String = "",          // 课程名称
    val teacher: String = "",       // 授课教师
    val classroom: String = "",     // 教室
    val dayOfWeek: Int = 1,        // 星期几（1=周一 ... 7=周日）
    val startTime: String = "08:00",// 开始时间
    val endTime: String = "09:35", // 结束时间
    val startWeek: Int = 1,        // 起始周
    val endWeek: Int = 20,         // 结束周
    val color: Int = ...            // 颜色
    val notes: String? = null,      // 备注
    val weekType: String = "ALL",   // 每周/单周/双周
    val orderIndex: Int = 0
)
```

**讲解要点：**
- `@Entity` 注解 → Room 自动建表
- **外键** (`ForeignKey`) → Course 关联到 Schedule，删除课表时自动删除课程
- **索引** → 加速按 scheduleId 查询
- `isInWeek(week)` 方法 → 判断课程是否在指定周出现

#### 4. DAO 数据访问

**`CourseDao.kt`：**
```kotlin
@Dao
interface CourseDao {
    // Flow 返回 → 数据变化时 UI 自动更新
    @Query("SELECT * FROM courses WHERE scheduleId = :scheduleId ORDER BY dayOfWeek, startTime")
    fun getCoursesBySchedule(scheduleId: Long): Flow<List<Course>>

    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insert(course: Course): Long

    @Update
    suspend fun update(course: Course)

    @Delete
    suspend fun delete(course: Course)
}
```

**讲解要点：**
- `@Dao` → 数据访问对象
- **Flow 返回值** → 响应式查询，数据库变化时自动推送新数据
- `suspend` → 协程挂起函数，不阻塞主线程

**`OCRProcessor.kt`：**
```kotlin
class OCRProcessor {
    // 使用 ML Kit 拉丁文字识别器
    private val recognizer = TextRecognition.getClient(TextRecognizerOptions.DEFAULT_OPTIONS)

    data class OCRResult(
        val text: String,
        val blocks: List<TextBlock> = emptyList()
    )

    suspend fun processBitmap(bitmap: Bitmap): OCRResult {
        val image = InputImage.fromBitmap(bitmap, 0)

        // process() 是异步的，用 .await() 转协程
        val visionText = recognizer.process(image).await()
        val text = visionText.text

        return OCRResult(text = text, blocks = /* 位置信息 */)
    }
}
```

**讲解要点：**

#### 2. StateFlow 响应式数据

```kotlin
// 当前激活的课表
private val _activeSchedule = MutableStateFlow<Schedule?>(null)
val activeSchedule: StateFlow<Schedule?> = _activeSchedule.asStateFlow()

// 课表变化时，自动查询对应的课程列表
val courses: StateFlow<List<Course>> = _activeSchedule.flatMapLatest { schedule ->
    if (schedule != null) {
        courseDao.getCoursesBySchedule(schedule.id)
    } else {
        flowOf(emptyList())
    }
}.stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())
```

**讲解要点：**
- `StateFlow` → 有状态的数据流，UI 通过 `collectAsState()` 订阅

```kotlin
private fun computeCurrentWeek(): Int {
    val semesterStart = if (month >= Calendar.SEPTEMBER) {
        // 秋季学期从 9 月 1 日开始
        Calendar.getInstance().apply { set(year, Calendar.SEPTEMBER, 1) }
    } else {
        // 春季学期从 2 月 15 日开始
        Calendar.getInstance().apply { set(year, Calendar.FEBRUARY, 15) }
    }
    val diffInMillis = calendar.timeInMillis - semesterStart.timeInMillis
    return (diffInMillis / (7 * 24 * 60 * 60 * 1000L)).toInt() + 1
}
```

```kotlin
    Row {

            Icon(Icons.AutoMirrored.Filled.KeyboardArrowLeft)
        }

        // 当前周数显示
        Column {
            Text(if (displayedWeek == currentWeek) "本周" else "第${displayedWeek}周")

        // ▶ 下一周

            Icon(Icons.AutoMirrored.Filled.KeyboardArrowRight)
        }
    }

    // 不在本周时显示"回到本周"按钮
    if (displayedWeek != currentWeek) {
        OutlinedButton(onClick = onGoToCurrentWeek) {
            Text("回到本周")
        }
    }
}
```
